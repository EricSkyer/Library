private void Bs(int[] a)
{
    for (int i = 0; i < a.Length; i++)
    {
        for (int j = 1; j < a.Length; j++)
        {
            if (a[j - 1] < a[j])
            {
                int temp = a[j - 1];
                a[j - 1] = a[j];
                a[j] = temp;
            }
        }
    }
}
